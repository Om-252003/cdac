/*

Even though you run "java myclass", the JVM first ensures that all parent classes are loaded and initialized before the child class.

So base is loaded first, then sub, and finally myclass.

*/

class base
{
	static
	{
		System.out.println("in base static");
	}
}
class sub extends base
{
	static
	{
		System.out.println("in sub static");
	}
}
public class myclass extends sub
{
	static
	{
		System.out.println("in myclass static");
	}
	public static void main(String args[])
	{
	}
}
