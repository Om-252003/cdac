			Connection_Provider.py

import mysql.connector

def getConnection():
    # Connect to MySQL
    conn = mysql.connector.connect(
        host="localhost",
        user="root",
        password="root",
        database="pythondb1"
    )
    return conn



create another python file (script)  e.g. First.py 


import mysql.connector
from Connection_Provider import *
conn=getConnection()

# Create a cursor
cursor = conn.cursor()

# Call stored procedure
cursor.callproc("GetEmployees")

# Fetch results (stored procedures return results as a result set in cursor.stored_results())
for result in cursor.stored_results():
    for row in result.fetchall():
        print(row)  # Each row is a tuple (id, name)

# Clean up
cursor.close()
conn.close()
