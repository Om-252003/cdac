import socket

host = "0.0.0.0"  # listen on all interfaces
port = 12345

s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
s.bind((host, port))
s.listen(1)
print("Server listening...")

conn, addr = s.accept()
print("Connected by:", addr)

while True:
    data = conn.recv(1024)
    if not data:
        break
    print("Client:", data.decode())
    msg = input("Server reply: ")
    conn.send(msg.encode())

conn.close()
