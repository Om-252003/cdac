"""
== → Value Equality

Checks whether the values are the same.

It calls the __eq__ method of the object.

Example:
"""

a = 10
b = 10
print(a == b)   # True, because both have value 10
print(a is b)   # True, because both a and b refers
                #    to the same object

"""
2. is → Identity Equality

Checks whether two variables point to the exact same object in memory.

Example:
"""

a = [1, 2, 3]
b = [1, 2, 3]

print(a == b)  # True  (values are same)
print(a is b)  # False (different objects in memory)
